## Github Action integration

```yaml
name: Deployment

on:
  push:
    tags:
      - "*"

jobs:
  build:
    name: Build
    runs-on: ubuntu-latest
    steps:
      - name: PromptPal Setup
        run: |
          wget https://github.com/PromptPal/cli/releases/download/v1.1.0/cli_Linux_x86_64.tar.gz
          tar -zxvf cli_Linux_x86_64.tar.gz
      - name: Gen
        env:
          PROMPTPAL_ENDPOINT: ${{ secrets.PROMPTPAL_ENDPOINT }}
          PROMPTPAL_API_TOKEN: ${{ secrets.PROMPTPAL_API_TOKEN }}
        run: |
          ./promptpal g
      - name: Build
	  	run: SOME_CUSTOM_BUILD_PROCESS
```
